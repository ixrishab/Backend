Select * from Departments;
Select * from Employees;

Select * from Customers c left join Orders o on o.customer_id = c.customer_id;

Select c.* from Customers c left join Orders o on o.customer_id = c.customer_id where o.order_id is null;

Select * from Departments d left join Employees e on d.dep_id = e.dep_id;

Select * from Departments left join Employees on Departments.dep_id = Employees.dep_id where Employees.emp_id is null;

Select * from Products p left join order_items oi on oi.product_id = p.product_id where oi.order_id is null;
