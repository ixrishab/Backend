TASK - 10 (join & having)



Select c.customer_id,c.name,COUNT(o.customer_id) as no_of_Orders from Orders o join Customers c on c.customer_id = o.customer_id group by  o.customer_id,c.customer_id,c.name having COUNT(o.customer_id) > 2;

Select d.dep_id,d.dep_name,COUNT(e.dep_id) as dep_emp_count from Employees e join Departments d on d.dep_id = e.dep_id group by e.dep_id,d.dep_id,d.dep_name having COUNT(e.dep_id) > 5;
 

 Select p.product_id,p.product_name,COUNT(oi.product_id) as count_product_ordered from order_items oi join  Products p on oi.product_id=p.product_id group by p.product_id,p.product_name,oi.product_id having COUNT(oi.product_id) >1;

 Select c.customer_id,c.name,SUM(amount) as sum_Order from Orders o join Customers c on c.customer_id=o.customer_id group by c.customer_id,c.name,amount having SUM(amount) > 1000;
