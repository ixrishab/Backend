TASK-4 (Full join)

Select * from Customers Full Outer join Orders on Customers.customer_id = Orders.customer_id;

Select * from Employees full join Departments on Employees.dep_id = Departments.dep_id;

Select * from Employees left join Departments on Employees.dep_id = Departments.dep_id
UNION
Select * from Employees right join Departments on Departments.dep_id = Employees.dep_id;
