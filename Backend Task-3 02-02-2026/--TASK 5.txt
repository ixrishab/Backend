--TASK 5 (CouNt)
Select Count(*) as CountOfTotalStudents from STUDENTS;
Select Count(student_id) as CountOfActiveStudents from STUDENTS where active_flag = 1;

--(Sum /Avg)
 Select sum(fees) as TotalCourseFees from COURSES;
 Select Avg(fees) as AverageCourseFees from COURSES;
 Select sum(fees) from COURSES where fees > 20000;

 --(Min / Max)
 Select min(age) from STUDENTS;
 Select Max(fees) from COURSES;
select max(registration_date) from STUDENTS;
select min(registration_date) from STUDENTS;

--Group By
Select active_flag, count(active_flag) as active_flag_count from STUDENTS group by active_flag;
Select age,COUNT(age) from STUDENTS group by age;

--Having
select course_id, count(*) as  enrollment_count from ENROLMENTS group by course_id having count(*) > 3;
select avg(age) as average_age from STUDENTS group by age having age > 21;
SELECT course_id, COUNT(*) AS total_enrollments FROM ENROLMENTS GROUP BY course_id HAVING COUNT(*) > 2;
